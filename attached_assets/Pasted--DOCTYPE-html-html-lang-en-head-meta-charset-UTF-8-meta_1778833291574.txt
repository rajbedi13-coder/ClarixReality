// ── RENDER FEED ───────────────────────────────────────────
function renderFeed() {
  const grid = document.getElementById("feed-grid");
  const empty = document.getElementById("empty-state");
  const heading = document.getElementById("feed-heading");
  const nl = document.getElementById("newsletter-block");

  let articles = ARTICLES.filter(a => {
    const catOk = currentCategory === "all" || a.cat === currentCategory;
    const srchOk = !currentSearch || a.headline.toLowerCase().includes(currentSearch.toLowerCase()) || a.summary.toLowerCase().includes(currentSearch.toLowerCase());
    const savedOk = currentView !== "saved" || savedIds.has(a.id);
    return catOk && srchOk && savedOk;
  });

  // Newsletter visibility
  nl.style.display = (currentView === "home" && currentCategory === "all" && !currentSearch) ? "block" : "none";

  // Heading visibility
  if (currentCategory !== "all" || currentSearch || currentView === "saved") {
    heading.style.display = "block";
    let txt = currentSearch ? `Results for "${currentSearch}"` :
              currentView === "saved" ? "Saved Articles" :
              document.querySelector(".cat-btn.active span:last-child")?.textContent || "Articles";
    document.getElementById("feed-heading-text").textContent = txt;
    document.getElementById("feed-heading-count").textContent = `${articles.length} brief${articles.length !== 1 ? "s" : ""} found`;
  } else {
    heading.style.display = "none";
  }

  if (articles.length === 0) {
    grid.innerHTML = "";
    empty.style.display = "block";
    document.getElementById("empty-msg").textContent =
      currentView === "saved" ? "No saved articles yet. Click ◇ on any brief to save it." :
      currentSearch ? `No results for "${currentSearch}". Try a different keyword.` :
      "No articles in this category yet.";
    return;
  }
  empty.style.display = "none";

  const featured = (currentView === "home" && currentCategory === "all" && !currentSearch)
    ? articles.filter(a => a.featured) : [];
  const regular = articles.filter(a => !featured.includes(a));

  grid.innerHTML = [...featured, ...regular].map((a, i) => cardHTML(a, i, featured.includes(a))).join("");
}

}
